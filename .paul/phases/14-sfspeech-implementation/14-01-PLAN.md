    ```python
    def _run_sfspeech(audio: np.ndarray, model: str) -> str:
        import soundfile as sf

        locale = NSLocale.alloc().initWithLocaleIdentifier_("en-US")
        if model not in _sfspeech_recognizer_cache:
            _sfspeech_recognizer_cache[model] = SFSpeechRecognizer.alloc().initWithLocale_(locale)
        recognizer = _sfspeech_recognizer_cache[model]

        tmp_path = None
        transcript = ""
        try:
            with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as tmp:
                tmp_path = tmp.name
            sf.write(tmp_path, audio, 16000, subtype="PCM_16")

            url = NSURL.fileURLWithPath_(tmp_path)
            request = SFSpeechURLRecognitionRequest.alloc().initWithURL_(url)
            request.setRequiresOnDeviceRecognition_(True)
            request.setAddsPunctuation_(True)

            done = [False]
            result_holder = [None]

            def handler(result, error):
                if result is not None and result.isFinal():
                    result_holder[0] = result.bestTranscription().formattedString()
                    done[0] = True
                elif error is not None:
                    done[0] = True

            recognizer.recognitionTaskWithRequest_resultHandler_(request, handler)

            deadline = NSDate.dateWithTimeIntervalSinceNow_(5.0)
            while not done[0]:
                NSRunLoop.currentRunLoop().runMode_beforeDate_(
                    "NSDefaultRunLoopMode", NSDate.dateWithTimeIntervalSinceNow_(0.05)
                )
                if NSDate.date().compare_(deadline) != -1:
                    break

            transcript = result_holder[0] or ""
        finally:
            if tmp_path and os.path.exists(tmp_path):
                os.unlink(tmp_path)

        return transcript.strip()
    ```

    **WAV temp-file overhead (not a bottleneck):** The concern about SFSpeech suffering parakeet's overhead problem is worth addressing. Parakeet's original slowness was `from_pretrained()` running on every keypress (~5s) â€” NOT the WAV write. The WAV write (`sf.write`, float32â†’PCM16, 5s clip = ~160KB) takes ~1â€“5ms on SSD regardless of backend. SFSpeech reads that file via the system Speech framework (no ffmpeg), which is faster than parakeet's ffmpeg path. The `_sfspeech_recognizer_cache` eliminates any equivalent of `from_pretrained()` overhead.
    If profiling shows WAV I/O is surprisingly slow, the mitigation is `SFSpeechAudioBufferRecognitionRequest` + `AVAudioPCMBuffer` (skips disk entirely), but this requires more complex PyObjC code and was explicitly deferred in Phase 13 research. Do not implement the buffer path in this plan.

    Avoid: importing SFSpeech objects inside `_run_sfspeech` â€” they are loaded at module level via `objc.loadBundle`. The `objc.loadBundle` call injects `SFSpeechRecognizer`, `SFSpeechURLRecognitionRequest` etc. into `globals()` at import time.
