    ```python
    """Snippet expansion — maps spoken shorthand to predefined text."""
    from __future__ import annotations

    import sys
    from pathlib import Path

    try:
        import tomllib
    except ImportError:
        import tomli as tomllib  # Python < 3.11 fallback


    _CONFIG_PATH = Path.home() / ".config" / "local-whisper" / "snippets.toml"


    def load(path: Path = _CONFIG_PATH) -> dict[str, str]:
        """Load snippets from TOML config file.

        Returns empty dict if file does not exist.

        Args:
            path: Path to snippets.toml. Defaults to ~/.config/local-whisper/snippets.toml.

        Returns:
            Dict mapping trigger phrases to expansion text.
        """
        if not path.exists():
            return {}
        try:
            with path.open("rb") as f:
                data = tomllib.load(f)
            return {k.lower(): v for k, v in data.get("snippets", {}).items()}
        except Exception as exc:
            print(f"[local-whisper] Failed to load snippets: {exc}", file=sys.stderr)
            return {}


    def expand(text: str, snippets: dict[str, str]) -> str:
        """Expand text if it matches a snippet trigger (case-insensitive, full match).

        Args:
            text: Transcribed text to check.
            snippets: Loaded snippets dict (keys already lowercased).

        Returns:
            Expanded value if match found, original text otherwise.
        """
        return snippets.get(text.strip().lower(), text)
    ```

    Note: `tomllib` is stdlib in Python 3.11+. Project requires >=3.11, so no new dep needed.

    ```python
    def _record_and_process(self) -> None:
        try:
            audio_data: np.ndarray = audio.record_until_event(self._stop_event)
            if audio_data.size == 0:
                print("[local-whisper] No audio captured.", file=sys.stderr)
                return
            text = transcribe.run(audio_data)
            if not text:
                print("[local-whisper] Empty transcription.", file=sys.stderr)
                return
            text = snippets.expand(text, self._snippets)
            clipboard.write_and_paste(text)
        except Exception as exc:  # noqa: BLE001
            print(f"[local-whisper] Error: {exc}", file=sys.stderr)
        finally:
            self._recording = False
            if self._overlay:
                self._overlay.hide()
    ```

    ```python
    """Tests for snippets.load() and snippets.expand()."""
    from pathlib import Path

    import pytest

    from local_whisper.snippets import expand, load


    def test_load_returns_empty_when_file_missing(tmp_path: Path) -> None:
        assert load(tmp_path / "nonexistent.toml") == {}


    def test_load_parses_snippets_section(tmp_path: Path) -> None:
        config = tmp_path / "snippets.toml"
        config.write_text('[snippets]\n"calendly link" = "https://cal.com/user"\n')
        result = load(config)
        assert result == {"calendly link": "https://cal.com/user"}


    def test_load_lowercases_keys(tmp_path: Path) -> None:
        config = tmp_path / "snippets.toml"
        config.write_text('[snippets]\n"My Email" = "user@example.com"\n')
        result = load(config)
        assert "my email" in result


    def test_load_returns_empty_on_malformed_toml(tmp_path: Path) -> None:
        config = tmp_path / "snippets.toml"
        config.write_text("not valid toml ][")
        assert load(config) == {}


    def test_expand_returns_snippet_on_exact_match() -> None:
        snippets = {"calendly link": "https://cal.com/user"}
        assert expand("calendly link", snippets) == "https://cal.com/user"


    def test_expand_is_case_insensitive() -> None:
        snippets = {"my email": "user@example.com"}
        assert expand("My Email", snippets) == "user@example.com"
        assert expand("MY EMAIL", snippets) == "user@example.com"


    def test_expand_strips_whitespace_before_matching() -> None:
        snippets = {"calendly link": "https://cal.com/user"}
        assert expand("  calendly link  ", snippets) == "https://cal.com/user"


    def test_expand_returns_original_when_no_match() -> None:
        assert expand("hello world", {}) == "hello world"
        assert expand("hello world", {"other key": "value"}) == "hello world"
    ```
